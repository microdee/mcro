40 /**
41 * @brief A base class for a structured error handling and reporting with modular architecture and fluent API.
42 *
43 * @important
44 * Instantiate errors only with `IError::Make(new FMyError())` this ensures the minimal runtime reflection features.
45 *
46 * Many times runtime errors are unavoidable and if an API only gives indication of success or failure (let's say in
47 * the form of a boolean) that will be quite frustrating for the user to report, as it gives no direction of course
48 * what went wrong, how it went wrong, and when it went wrong. Slightly better when the API gives a list of things
49 * what can go wrong and return an item from that list when things go wrong. This of course still doesn't allow to
50 * provide much of a context for the user.
51 *
52 * An 'improvement' to that is using C++ exceptions, however it is not unanimously well received in the community
53 * because it can hide the fact that the API can bail on its caller. So when exceptions are enabled one may call
54 * every function of an API like if they were walking on a minefield. For this (and a couple more) reasons C++
55 * exceptions are disabled by default in Unreal projects and viewed as terrible practice to introduce it ourselves.
56 *
57 * Unreal does provide though the `TValueOrError` template which allows API's to indicate that they can fail in some
58 * ways without the need to consult an external documentation. It gives the developer total freedom however of what
59 * the error can be, so on its own it does not solve the questions of what/when/how.
60 *
61 * Using `TMaybe` with IError can be a powerful tool in the developer's arsenal when creating a library.
62 * IError can standardize a detailed and structured way of communicating errors without hindering call-site
63 * usage. It can also automate the method and the format of logging the (many times excessive amount of)
64 * information surrounding an error, or decide how it may be presented for the user.
65 */

555 /**
556 * @brief
557 * A `TValueOrError` alternative for IError which allows implicit conversion from values and errors (no need for
558 * `MakeError` or `MakeValue`) and is boolean testable. It also doesn't have ambiguous state such as
559 * `TValueOrError` has, so a TMaybe will always have either an error or a value, it will never have neither of
560 * them or both of them.
561 */
562 template <CNonVoid T>
563 struct TMaybe
564 {
565 using ValueType = T;
566
567 /**
568 * @brief
569 * Default initializing a TMaybe while its value is not default initializable, initializes the resulting
570 * TMaybe to an erroneous state.
571 */
572 template <typename = T>
573 requires (!CDefaultInitializable<T>)
574 TMaybe() : Error(IError::Make(new FUnavailable())
575 ->WithMessageF(
576 TEXT_"TMaybe has been default initialized, but a Value of {0} cannot be default initialized",
577 TTypeName<T>
578 )
579 ) {}
580
581 /** @brief If T is default initializable then the default state of TMaybe will be the default value of T, and not an error */
582 template <CDefaultInitializable = T>
583 TMaybe() : Value(T{}) {}
584
585 /** @brief Enable copy constructor for T only when T is copy constructable */
586 template <CConvertibleToDecayed<T> From, CCopyConstructible = T>
587 TMaybe(From const& value) : Value(value) {}
588
589 /** @brief Enable move constructor for T only when T is move constructable */
590 template <CConvertibleToDecayed<T> From, CMoveConstructible = T>
591 TMaybe(From&& value) : Value(Forward<From>(value)) {}
592
593 /** @brief Enable copy constructor for TMaybe only when T is copy constructable */
594 template <CConvertibleToDecayed<T> From, CCopyConstructible = T>
595 TMaybe(TMaybe<From> const& other) : Value(other.Value) {}
596
597 /** @brief Enable move constructor for TMaybe only when T is move constructable */
598 template <CConvertibleToDecayed<T> From, CMoveConstructible = T>
599 TMaybe(TMaybe<From>&& other) : Value(MoveTemp(other.Value)) {}
600
601 /** @brief Set this TMaybe to an erroneous state */
602 template <CError ErrorType>
603 TMaybe(TSharedRef<ErrorType> const& error) : Error(error) {}
604
605 bool HasValue() const { return Value.IsSet(); }
606 bool HasError() const { return Error.IsValid(); }
607
608 auto TryGetValue() -> TOptional<T>& { return Value; }
609 auto TryGetValue() const -> TOptional<T> const& { return Value; }
610
611 auto GetValue() -> T& { return Value.GetValue(); }
612 auto GetValue() const -> T const& { return Value.GetValue(); }
613
614 auto GetError() const -> IErrorPtr { return Error; }
615 auto GetErrorRef() const -> IErrorRef { return Error.ToSharedRef(); }
616
617 operator bool() const { return HasValue(); }
618
619 operator TValueOrError<T, IErrorPtr>() const
620 {
621 if (HasValue())
622 return MakeValue(Value.GetValue());
623 return MakeError(Error);
624 }
625
626 private:
627 TOptional<T> Value;
628 IErrorPtr Error;
629 };
