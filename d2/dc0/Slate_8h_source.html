81 /**

83 * An attribute block functor which takes in reference of FArguments or FSlotArguments and returns the
84 * same reference but presumably setting some Slate attributes before that. This is useful for modularizing the
85 * Slate declarative syntax.

87 * Use `TAttributeBlock` alias for your functions for better convenience

89 template <CWidgetOrSlotArguments T>
90 struct TAttributeBlockFunctor

92 using Function = TFunction<T&(T&)>;

94 template <CConvertibleToDecayed<Function> Arg>
95 TAttributeBlockFunctor(Arg&& function) : Storage(Forward<Arg>(function)) {}

97 Function Storage;
98
99 T& operator () (T& args) const
100 {
101 return Storage(args);

104 /**
105 * @brief
106 * The "append attribute block" operator which allows pre-defined "blocks of slate attributes" naturally fit
107 * inside the Slate declarative syntax. Traditionally repeated structures in Slate were expressed as either
108 * explicit mutations on widgets after they were created or as entirely separate compound widgets. Either way
109 * breaks the flow of the declarative syntax and makes using Slate sometimes pretty clunky. This operator aims
110 * to make widget composition more comfortable.
111 *
112 * @tparam Arguments Right hand side FArguments or FSlotArguments
113 * @param args r-value reference right hand side FArguments or FSlotArguments
114 * @param attributes An attribute block function
115 * @return The same reference as args or a new slot if that has been added inside the attribute block
116 */
117 friend T& operator / (TIdentity_T<T>&& args, TAttributeBlockFunctor const& attributes)
118 {
119 return attributes(args);
120 }

154 /**
155 * @brief Add multiple slots at the same time with the declarative syntax derived from an input data array.
156 *
157 * @code
158 * void SMyWidget::Construct(const FArguments& args)
159 * {
160 * using namespace Mcro::Slate;
161 *
162 * ChildSlot
163 * [
164 * SNew(SVerticalBox)
165 * + TSlots(args._DataArray, [](const FMyData& data)
166 * {
167 * FText dataText = FText::FromString(data.ToString());
168 * return SVerticalBox::Slot()
169 * . HAlign(HAlign_Fill)
170 * . AutoHeight()
171 * [
172 * SNew(STextBlock)
173 * . Text(dataText)
174 * ];
175 * })
176 * + SVerticalBox::Slot()
177 * . HAlign(HAlign_Fill)
178 * . AutoHeight()
179 * [
180 * SNew(STextBlock)
181 * . Text(INVTEXT_"Footer after the list of data")
182 * ]
183 * ];
184 * }
185 * @endcode
186 */
187 template <
188 CRangeMember Range,
189 CFunctionLike Transform,
190 CFunctionLike OnEmpty = TUniqueFunction<TFunction_Return<Transform>()>,
191 CSlotArguments = TFunction_Return<Transform>
192 >
193 requires (TFunction_ArgCount<Transform> == 1)
194 struct TSlots
195 {
196 TSlots(Range const& range, Transform&& transform, TOptional<OnEmpty>&& onEmpty = {})
197 : RangeRef(const_cast<Range&>(range))
198 , TransformStorage(MoveTemp(transform))
199 , OnEmptyStorage(MoveTemp(onEmpty))
200 {}
201 TSlots(Range& range, Transform&& transform, TOptional<OnEmpty>&& onEmpty = {})
202 : RangeRef(range)
203 , TransformStorage(MoveTemp(transform))
204 , OnEmptyStorage(MoveTemp(onEmpty))
205 {}
206
207 TSlots(TSlots const&) = delete;
208 TSlots(TSlots&& o) noexcept
209 : RangeRef(o.RangeRef)
210 , TransformStorage(MoveTemp(o.TransformStorage))
211 , OnEmptyStorage(MoveTemp(o.OnEmptyStorage))
212 {
213 }
214
215 TSlots& operator=(TSlots const&) = delete;
216 TSlots& operator=(TSlots&& o) noexcept
217 {
218 if (this == &o)
219 return *this;
220 RangeRef = o.RangeRef;
221 TransformStorage = MoveTemp(o.TransformStorage);
222 OnEmptyStorage = MoveTemp(o.OnEmptyStorage);
223 return *this;
224 }
225
226 template <CWidgetArguments Arguments>
227 void Append(Arguments& args)
228 {
229 using namespace Mcro::Range;
230 if (IteratorEquals(RangeRef.begin(), RangeRef.end()) && OnEmptyStorage.IsSet())
231 {
232 args + OnEmptyStorage.GetValue()();
233 return;
234 }
235
236 for (auto it = RangeRef.begin(); !IteratorEquals(it, RangeRef.end()); ++it)
237 args + TransformStorage(*it);
238 }
239
240 /** @copydoc TSlots */
241 template <CWidgetArguments Arguments>
242 friend Arguments& operator + (Arguments& args, TSlots&& slots)
243 {
244 slots.Append(args);
245 return args;
246 }
247
248 private:
249 Range& RangeRef;
250 Transform TransformStorage;
251 TOptional<OnEmpty> OnEmptyStorage;
252 };
