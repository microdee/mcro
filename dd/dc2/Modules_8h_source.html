47 /** @brief Use this in global variables to automatically do things on module startup or shutdown */
48 template <CDerivedFrom<IObservableModule> M>
49 struct TObserveModule
50 {
51 /**
52 * @brief
53 * Default constructor will try to infer module name from type name. Given convention
54 * `(F|I)Foobar(Module(Interface)?)?` the extracted name will be `Foobar`. If your module doesn't follow this
55 * naming use the constructor accepting an FName
56 */
57 TObserveModule()
58 {
59 auto moduleName = TTypeString<M>().Mid(1);
60 moduleName.RemoveFromEnd(TEXT_"Module");
61 moduleName.RemoveFromEnd(TEXT_"ModuleInterface");
62 ObserveModule(moduleName);
63 }
64
65 /** @brief This constructor provides an explicit FName for getting the module */
66 TObserveModule(FName const& moduleName)
67 {
68 ObserveModule(moduleName);
69 }
70
71 /**
72 * @brief
73 * Event broadcasted on module startup or immediately executed upon subscription if module has already been
74 * started up.
75 */
76 TBelatedEventDelegate<void()> OnStartupModule;
77
78 /**
79 * @brief
80 * Event broadcasted on module shutdown or immediately executed upon subscription if module has already been
81 * shut down.
82 */
83 TBelatedEventDelegate<void()> OnShutdownModule;
84
85 /** @brief Specify function to be executed on startup */
86 TObserveModule& OnStartup(TFunction<void()>&& func)
87 {
88 OnStartupModule.Add(InferDelegate::From(func));
89 return *this;
90 }
91
92 /** @brief Specify function to be executed on shutdown */
93 TObserveModule& OnShutdown(TFunction<void()>&& func)
94 {
95 OnShutdownModule.Add(InferDelegate::From(func));
96 return *this;
97 }
98
99 private:
100 M* Module = nullptr;
101
102 void ObserveModule(FName const& moduleName)
103 {
104 decltype(auto) manager = FModuleManager::Get();
105 M* module = static_cast<M*>(manager.GetModule(moduleName));
106 if (!module)
107 {
108 manager.OnModulesChanged().AddLambda([this, moduleName](FName name, EModuleChangeReason changeReason)
109 {
110 if (changeReason == EModuleChangeReason::ModuleLoaded && moduleName == name)
111 ObserveModule(moduleName);
112 });
113 }
114 else
115 {
116 Module = module;
117 module->OnStartupModule.Add(OnStartupModule.Delegation());
118 module->OnShutdownModule.Add(OnShutdownModule.Delegation());
119 }
120 }
121 };
122}
