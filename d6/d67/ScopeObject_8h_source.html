19 /**
20 * @brief
21 * A struct to emulate regular C++ RAII object behavior with UObjects. When this struct is instantiated the given
22 * object type is also created. An object wrapped in this struct is never invalid, and doesn't get garbage
23 * collected until it's in scope.
24 *
25 * It is safe to create this struct on any thread, GC is deferred until constructor is finished. The underlying
26 * storage is a `TStrongObjectPtr`.
27 *
28 * If the given object type happen to have an `Intialize` member method, that's also called with the extra
29 * arguments provided in the constructor.
30 */
31 template <CUObject T>
32 struct TScopeObject
33 {
34 template <typename... Args>
35 TScopeObject(FConstructObjectParameters&& params, Args&&... args)
36 {
37 FGCScopeGuard avoidGC;
38 Storage.Reset(NewInit<T>(Forward<FConstructObjectParameters>(params), Forward<Args>(args)...));
39 }
40
41 const T* operator -> () const { return Storage.Get(); }
42 T* operator -> () { return Storage.Get(); }
43
44 operator const T* () const { return Storage.Get(); }
45 operator T* () { return Storage.Get(); }
46
47 T const& Get() const { return *Storage.Get(); }
48 T& Get() { return *Storage.Get(); }
49
50 private:
51 TStrongObjectPtr<T> Storage;
52 };
53}
