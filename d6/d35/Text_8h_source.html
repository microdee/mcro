132 template <
133 typename CharFrom, typename CharOutput,
134 typename StringFrom,
135 CFunctionLike PtrGetter, CFunctionLike LengthGetter, CFunctionLike Constructor,
136 typename StringTo = TFunction_Return<Constructor>
137 >
138 StringTo HighLevelStringCast(const StringFrom& strIn, PtrGetter&& getPtr, LengthGetter&& getLength, Constructor&& construct)
139 {
140 if constexpr (CSameAs<CharFrom, CharOutput>)
141 {
142 return construct(
143 reinterpret_cast<const CharOutput*>(getPtr()),
144 getLength()
145 );
146 }
147 else
148 {
149 auto conversion = StringCast<CharOutput>(
150 reinterpret_cast<const CharFrom*>(getPtr()),
151 getLength()
152 );
153 return construct(
154 reinterpret_cast<const CharOutput*>(conversion.Get()),
155 conversion.Length()
156 );
157 }
158 }

377 /**
378 * @brief Attempt to convert anything to string which can tell via some method how to do so
379 *
380 * This may be more expensive than directly using an already existing designated string conversion for a given
381 * type, because it uses `FStringFormatArg` and a `TAsFormatArgument` as intermediate steps. However, this can be
382 * still useful for types where such conversion doesn't already exist or when using this in a template.
383 *
384 * It's still much faster than using `FMT_(myVar) "{0}"`
385 */
386 template <CStringFormatArgument T>
387 requires(!CSameAsDecayed<T, FString>)
388 FString AsString(T&& input)
389 {
390 FStringFormatArg format(AsFormatArgument(input));
391 return MoveTemp(format.StringValue);
