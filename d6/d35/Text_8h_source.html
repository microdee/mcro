120 template <
121 typename CharFrom, typename CharOutput,
122 typename StringFrom,
123 CFunctionLike PtrGetter, CFunctionLike LengthGetter, CFunctionLike Constructor,
124 typename StringTo = TFunction_Return<Constructor>
125 >
126 StringTo HighLevelStringCast(const StringFrom& strIn, PtrGetter&& getPtr, LengthGetter&& getLength, Constructor&& construct)
127 {
128 if constexpr (CSameAs<CharFrom, CharOutput>)
129 {
130 return construct(
131 reinterpret_cast<const CharOutput*>(getPtr()),
132 getLength()
133 );
134 }
135 else
136 {
137 auto conversion = StringCast<CharOutput>(
138 reinterpret_cast<const CharFrom*>(getPtr()),
139 getLength()
140 );
141 return construct(
142 reinterpret_cast<const CharOutput*>(conversion.Get()),
143 conversion.Length()
144 );
145 }
146 }

319 /**
320 * @brief Attempt to convert anything to string which can tell via some method how to do so
321 *
322 * This may be more expensive to directly use than an already existing designated string conversion for a given
323 * type, because it uses `FStringFormatArg` and `% FStringFormatTag()` as intermediate steps. However, this can be
324 * still useful for types where such conversion doesn't already exist or when using this in a template.
325 *
326 * It's still much faster than using `FMT_(myVar) "{0}"`
327 */
328 template <CStringFormatArgument T>
329 FString AsString(T&& input)
330 {
331 FStringFormatArg format(input % FStringFormatTag());
332 return MoveTemp(format.StringValue);
333 }

335 /**
336 * @brief Create an ordered argument list for a string format from input arguments
337 *
338 * While you can it is not recommended to be used directly because the boilerplate it still needs is very verbose.
339 * Check `_FMT` or `FMT_` macros for a comfortable string format syntax.
340 */
341 template <CStringFormatArgument... Args>
342 FStringFormatOrderedArguments OrderedArguments(Args&&... args)
