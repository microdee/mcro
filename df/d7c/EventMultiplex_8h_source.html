18/**
19 * @brief
20 * Glue together a regular function, a native event delegate and a dynamic multicast delegate, so a given API
21 * can serve
22 * - events via function override
23 * - C++ listeners via a native event delegate
24 * - Blueprint listeners via a dynamic multicast delegate
25 * with calling only a single class member.
26 *
27 * @param functionName The base name of the function which can be listened to with multiplexing
28 */
29#define MCRO_DYNAMIC_EVENT_MULTIPLEX(functionName) \
30 Mcro::Delegates::TNativeMultiplex<F##functionName> functionName##Multiplex \
31 { \
32 Mcro::Delegates::From(this, &MCRO_THIS_TYPE::functionName), \
33 Mcro::Delegates::From(functionName##Event) \
34 } \
35
36/**
37 * In case the multiplexed function returns a value instead of just returning void, add this macro below the dynamic
38 * multicast delegate declaration
39 *
40 * @param delegate The name of the dynamic multicast delegate
41 * @param type The return type of the multiplexed function
42 */
43#define MCRO_DYNAMIC_RETURN(delegate, type) \
44 template<> struct Mcro::Delegates::TReturnOverride_Struct<delegate> { using Type = type; }
