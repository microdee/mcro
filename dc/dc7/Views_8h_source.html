79 /**
80 * @brief
81 * Return true if input ranges match their values and their order.
82 *
83 * This runs in O(N) time when result is true, unless both input ranges are `CCountableRange`, `matchOnlyBeginning`
84 * is false and the two input ranges have different size.
85 *
86 * @param left Input range
87 * @param right Range to compare with
88 * @param matchOnlyBeginning
89 * By default MatchOrdered returns false for ranges with different lengths
90 */
91 template <CRangeMember Left, CRangeMember Right>
92 requires CConvertibleToDecayed<TRangeElementType<Right>, TRangeElementType<Left>>
93 bool MatchOrdered(Left&& left, Right&& right, bool matchOnlyBeginning = false)
94 {
95 if (IsEmpty(left) && IsEmpty(right)) return true;
96
97 if constexpr (CCountableRange<Left> && CCountableRange<Right>)
98 if (!matchOnlyBeginning && size(left) != size(right))
99 return false;
100
101 auto leftIt = left.begin();
102 auto rightIt = right.begin();
103 for (;;)
104 {
105 if (IteratorEquals(leftIt, left.end()) || IteratorEquals(rightIt, right.end()))
106 return matchOnlyBeginning || (
107 IteratorEquals(leftIt, left.end()) && IteratorEquals(rightIt, right.end())
108 );
109
110 if (*leftIt != *rightIt) return false;
111 ++leftIt;
112 ++rightIt;
113 }
114 }
