41 /**
42 * @brief
43 * Inherit via this template to allow other API to reflect upon the base types of deriving class. Base types are
44 * inherited as public. If you want privately inherited base classes, just inherit them as normal.
45 *
46 * Usage:
47 * @code
48 * class FMyThing : public TInherit<IFoo, IBar, IEtc>
49 * {
50 * // ...
51 * }
52 * @endcode
53 */
54 template <typename... BaseTypes>
55 class TInherit : public BaseTypes...
56 {
57 public:
58 using Bases = TTypes<BaseTypes...>;
59 };

94 /**
95 * @brief
96 * Give the opportunity to customize object lifespan operations for `FAny` by either specializing this template
97 * or just providing functors in-place
98 *
99 * @tparam T The type being set for an FAny
100 */
101 template <typename T>
102 struct TAnyTypeFacilities
103 {
104 TFunction<void(T*)> Destruct {[](T* object)
105 {
106 delete object;
107 }};
108
109 TFunction<T*(T const&)> CopyConstruct {[](T const& object)
110 {
111 return new T(object);
112 }};
113
114 TFunction<T*(T&&)> MoveConstruct {[](T&& object)
115 {
116 return new T(Forward<T>(object));
117 }};
118 };

120 /**
121 * @brief
122 * A simplistic but type-safe and RAII compliant storage for anything. Enclosed data is owned by this type.
123 *
124 * Use this with care, the underlying data can be only accessed with the same type as it has been constructed with,
125 * or with types provided by `ValidAs`. This means derived classes cannot be accessed with their base types safely
126 * and implicitly. MCRO however provides a methods for classes to allow them exposing base types to FAny (and
127 * other facilities):
128 * @code
129 * class FMyThing : public TInherit<IFoo, IBar, IEtc>
130 * {
131 * // ...
132 * }
133 * @endcode
134 * `TInherit` has a member alias `using Bases = TTypes<...>` and that can be used by FAny to automatically register
135 * base classes as compatible ones.
136 *
137 * Enclosed value must be copy and move constructible and assignable.
138 *
139 * @todo
140 * C++ 26 has promising proposal for static value-based reflection, which can gather metadata from classes
141 * or even emit them. The best summary I found so far is a stack-overflow answer https://stackoverflow.com/a/77477029
142 * Once that's available we can gather base classes in compile time, and do dynamic casting of objects without
143 * the need for intrusive extra syntax, or extra work at construction.
144 * Currently GCC's `__bases` would be perfect for the job, but other popular compilers don't have similar
145 * intrinsics. Once such a feature becomes widely available base classes can be automatically added as aliases for
146 * types wrapped in FAny.
147 */
148 struct MCRO_API FAny
149 {
150 template <CCopyable T>
151 FAny(T* newObject, TAnyTypeFacilities<T> const& facilities = {})
152 : Storage(newObject)
153 , MainType(TTypeOf<T>)
154 , Destruct([facilities](FAny* self)
155 {
156 T* object = static_cast<T*>(self->Storage);
157 facilities.Destruct(object);
158 self->Storage = nullptr;
159 })
160 , CopyConstruct([facilities](FAny* self, FAny const& other)
161 {
162 const T* object = static_cast<const T*>(other.Storage);
163 self->Storage = facilities.CopyConstruct(*object);
164
165 CopyTypeInfo(self, &other);
166 })
167 , MoveConstruct([facilities](FAny* self, FAny&& other)
168 {
169 T& object = *static_cast<T*>(other.Storage);
170 self->Storage = facilities.MoveConstruct(MoveTemp(object));
171
172 CopyTypeInfo(self, &other);
173 })
174 {
175 ValidTypes.Add(MainType);
176
177 if constexpr (CHasBases<T>)
178 {
179 Detail::ForEachExplicitBase<typename T::Bases>([this] <typename Base> (TTypes<Base>&&)
180 {
181 AddAlias(TTypeOf<Base>);
182 });
183 }
184 }

188 FAny(FAny&& other);
189 ~FAny();

207 /** @brief Specify one type the enclosed value can be safely cast to, and is valid to be used with `TryGet`. */
208 template <typename T, typename Self>
209 decltype(auto) WithAlias(this Self&& self)
210 {
211 self.AddAlias(TTypeOf<T>);
212
213 if constexpr (CHasBases<T>)
214 {
215 Detail::ForEachExplicitBase<typename T::Bases>([&] <typename Base> (TTypes<Base>&&)
216 {
217 self.AddAlias(TTypeOf<Base>);
218 });
219 }
220 return Forward<Self>(self);
221 }
222
223 /** @brief Specify multiple types the enclosed value can be safely cast to, and are valid to be used with `TryGet`. */
224 template <typename Self, typename... T>
225 decltype(auto) With(this Self&& self, TTypes<T...>&&)
226 {
227 (self.AddAlias(TTypeOf<T>), ...);
228 return Forward<Self>(self);
229 }
230
231 FORCEINLINE bool IsValid() const { return static_cast<bool>(Storage); }
232 FORCEINLINE FType GetType() const { return MainType; }
233 FORCEINLINE TSet<FType> const& GetValidTypes() const { return ValidTypes; }
234
235 private:
236 void AddAlias(FType const& alias);
237 static void CopyTypeInfo(FAny* self, const FAny* other);

244 TFunction<void(FAny* self, FAny&& other)> MoveConstruct {};

246 TSet<FType> ValidTypes {};
247 };
