30 struct TFunctionMeta
31 {
32 static constexpr size_t ArgumentCount = sizeof...(Args);
33
34 using Return = ReturnIn;

37 /** The input parameters of the function as a tuple type. Types are not decayed. */

40 /** The input parameters of the function as a tuple type. Types are decayed (useful for storage) */

42

44 using Signature = Return(Args...);
45
46 template <int I>
47 using Arg = typename TTupleElement<I, Arguments>::Type;
48
49 template <int I>
50 using ArgDecay = typename TTupleElement<I, ArgumentsDecay>::Type;
51 };
