356 /**
357 * @brief
358 * Defers a set of arguments for a function call later with its first argument. This is useful for developing
359 * fluent API operators.
360 */
361 template <auto FuncPtr, CFunctionPtr Function = decltype(FuncPtr)>
362 requires (
363 CFunctionPtr<decltype(FuncPtr)>
364 && TFunction_ArgCount<decltype(FuncPtr)> > 0
365 )
366 struct TDeferFunctionArguments
367 {
368 using FirstArg = TFunction_Arg<Function, 0>;
369 using ExtraArgs = TSkip<1, TFunction_Arguments<Function>>;
370 using Return = TFunction_Return<Function>;
371
372 template <typename... Args>
373 TDeferFunctionArguments(Args... args)
374 : Storage(Forward<Args>(args)...)
375 {}
376
377 template <CConvertibleToDecayed<FirstArg> FirstArgRef>
378 Return operator () (FirstArgRef&& arg)
379 {
380 auto args = arg >> Storage;
381 return InvokeWithTuple<Function>(FuncPtr, args);
