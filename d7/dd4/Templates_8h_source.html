30 /**
31 * @brief Base struct containing traits of specified template (which only accepts type parameters)
32 *
33 * @warning
34 * Until this proposal https://www.open-std.org/jtc1/sc22/wg21/docs/papers/2020/p1985r0.pdf or equivalent is
35 * considered seriously, template traits only work with templates which only have type-parameters. Non-type
36 * parameters even when a default is specified for them will result in compile error.
37 */
38 template <template <typename...> typename Template>
39 struct TTemplate
40 {
41 template <typename T>
42 static constexpr bool Match = false;
43
44 template <typename... Params>
45 static constexpr bool Match<Template<Params...>> = true;
46
47 template <typename T>
48 static constexpr size_t ParameterCount = 0;
49
50 template <typename... Params>
51 static constexpr size_t ParameterCount<Template<Params...>> = sizeof...(Params);
52
53 template <typename T>
54 struct Parameters
55 {
56 using Type = TTuple<>;
57 };
58
59 template <typename... Params>
60 struct Parameters<Template<Params...>>
61 {
62 using Type = TTuple<Params...>;
63 };
64
65 template <typename T>
66 struct ParametersDecay
67 {
68 using Type = TTuple<>;
69 };
70
71 template <typename... Params>
72 struct ParametersDecay<Template<Params...>>
73 {
74 using Type = TTuple<std::decay_t<Params>...>;
75 };
76
77 template <typename Instance, int I>
78 using Param = typename TTupleElement<I, typename Parameters<Instance>::Type>::Type;
79
80 template <typename Instance, int I>
81 using ParamDecay = typename TTupleElement<I, typename ParametersDecay<Instance>::Type>::Type;
82 };
