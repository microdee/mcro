31 /**
32 * @brief
33 * Auto Modular Features are a workflow with Modular Features where the developer doesn't have to rely on string
34 * identifiers. TAutoModularFeature template take care of naming the feature and introduces some common
35 * functionality, like getter functions and runtime validations.
36 *
37 * First a feature is defined with its interface class like so:
38 * @code
39 * class IMyModularFeature : public TAutoModularFeature<IMyModularFeature>
40 * {
41 * // ...
42 * }
43 * @endcode
44 * Then each implementation of this feature are defined like so:
45 * @code
46 * class FMyFeatureImplementation : public IMyModularFeature, public IFeatureImplementation
47 * {
48 * FMyFeatureImplementation()
49 * {
50 * // See the inline docs for why this needs to be done
51 * Register()
52 * }
53 * }
54 * @endcode
55 * Then instantiate the feature implementation when needed:
56 * @code
57 * class FMyModule
58 * {
59 * TPimplPtr<FMyFeatureImplementation> MyImplementation;
60 * }
61 *
62 * if (...)
63 * {
64 * MyImplementation = MakePimpl<FMyFeatureImplementation>();
65 * }
66 * @endcode
67 * To access the feature implementations then just use
68 * @code
69 * if (IMyModularFeature::ImplementationCount() > 0)
70 * {
71 * IMyModularFeature::Get().MyStuff();
72 * }
73 * @endcode
74 *
75 * Internally the feature name will be identical to the class name. In this case `IMyModularFeature` will register
76 * as "IMyModularFeature". Technically one can get it via
77 * @code
78 * IModularFeatures::Get().GetModularFeature<IMyModularFeature>(TEXT_"IMyModularFeature")
79 * @endcode
80 * but it is strongly discouraged for type safety and keeping code simple.
81 *
82 * For globally available features you may use just simply a global variable, or if it's important to have the
83 * owning module fully initialized, `TModuleBoundObject` is recommended.
84 * @code
85 * TModuleBoundObject<FFoobarModule, FMyFeatureImplementation> GMyFeatureImplementation;
86 * @endcode
87 *
88 * @remarks
89 * IMyModularFeature::FeatureName() and TTypeFName<FMyFeatureImplementation>() can be used for runtime
90 * comparison / validation.
91 *
92 * @tparam FeatureIn Curiously Recurring Template argument of the feature
93 */
94 template<typename FeatureIn>
95 class TAutoModularFeature : public IAutoModularFeature, public IModularFeature
96 {
97 public:
98 using Feature = FeatureIn;
99 using AutoModularFeature = TAutoModularFeature;
100
101 /** @brief This event is triggered when an implementation of this feature is created */
102 static auto OnRegistered() -> TBelatedEventDelegate<void(Feature*)>&
103 {
104 static TBelatedEventDelegate<void(Feature*)> event;
105 return event;
106 }
107
108 /** @brief Get the name of the feature */
109 static FORCEINLINE FName FeatureName()
110 {
111 return TTypeFName<Feature>();
112 }
113
114 /** @return The number of implementations created for this feature */
115 static FORCEINLINE int32 ImplementationCount()
116 {
117 return IModularFeatures::Get().GetModularFeatureImplementationCount(FeatureName());
118 }
119
120 /**
121 * @brief
122 * Get the first existing implementation of this feature. If there are no implementations a check will fail.
123 */
124 static FORCEINLINE Feature& Get()
125 {
126 return IModularFeatures::Get().GetModularFeature<Feature>(FeatureName());
127 }
128
129 /** @brief Get the first existing implementation of this feature. Return nullptr If there are no implementations. */
130 static FORCEINLINE Feature* TryGet(const int32 index)
131 {
132 return static_cast<Feature*>(IModularFeatures::Get().GetModularFeatureImplementation(FeatureName(), index));
133 }
134
135 /** @return An array of all implementations of this feature */
136 static FORCEINLINE TArray<Feature*> GetAll()
137 {
138 return IModularFeatures::Get().GetModularFeatureImplementations<Feature>(FeatureName());
139 }
140
141 /**
142 * @brief
143 * Call this function in implementation constructors.
144 *
145 * This is a necessary boilerplate to maintain polymorphism of implementations. Otherwise, if the native
146 * registration function would be called directly in TAutoModularFeature default constructor, virtual function
147 * overrides are not yet known, and "deducing this" is not meant for constructors.
148 *
149 * @tparam Implementation Derived type of the implementation
150 * @param self Pointer to implementation registering itself
151 */
152 template<typename Implementation> requires CDerivedFrom<Implementation, Feature>
153 void Register(this Implementation&& self)
154 {
155 UE_LOG(
156 LogAutoModularFeature, Log,
157 TEXT_"Registering %s as %s feature",
158 *TTypeString<Implementation>(),
159 *TTypeString<Feature>()
160 );
161 IModularFeatures::Get().RegisterModularFeature(FeatureName(), &self);
162 OnRegistered().Broadcast(&self);
163 }
164
165 virtual ~TAutoModularFeature()
166 {
167 IModularFeatures::Get().UnregisterModularFeature(FeatureName(), this);
168 }
169
170 /**
171 * @brief
172 * Get the first implementation once it is registered, or return the first implementation immediately if
173 * there's already one registered.
174 *
175 * @return A future completed when the first implementation becomes available, or there's already one
176 *
177 * @warning
178 * If no implementations were created throughout the entire execution of the program, it will crash on exit
179 * with an unfulfilled promise error. When using this function either handle this scenario early, or
180 * use `OnRegistered()` if your particular feature may never be created.
181 */
182 static FORCEINLINE TFuture<Feature*> GetBelated()
183 {
184 // Shared promise is required as delegate lambdas are copyable
185 TSharedRef<TPromise<Feature*>> promise = MakeShared<TPromise<Feature*>>();
186 TFuture<Feature*> result = promise->GetFuture();
187
188 OnRegistered().Add(InferDelegate::From([promise](Feature* feature) mutable
189 {
190 promise->SetValue(feature);
191 }), {.Once = true});
192
193 return result;
194 }
195 };
196}
