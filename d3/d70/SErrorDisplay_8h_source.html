24 /**
25 * @brief
26 * A modular feature which allows other modules to inject their own UI into the common error display widget.
27 *
28 * Of course these functions are only triggered when an error is using `SErrorDisplay` as its widget type, but
29 * errors may declare any other widget type to be used to represent themselves.
30 */
31 class MCRO_API IErrorDisplayExtension : public TAutoModularFeature<IErrorDisplayExtension>
32 {
33 public:
34 virtual bool SupportsError(IErrorRef const& error) { return true; };
35 virtual TSharedPtr<SWidget> PostSeverity(IErrorRef const& error) { return {}; };
36 virtual TSharedPtr<SWidget> PostMessage(IErrorRef const& error) { return {}; };
37 virtual TSharedPtr<SWidget> PostDetails(IErrorRef const& error) { return {}; };
38 virtual TSharedPtr<SWidget> PostCodeContext(IErrorRef const& error) { return {}; };
39 virtual TSharedPtr<SWidget> PostErrorPropagation(IErrorRef const& error) { return {}; };
40 virtual TSharedPtr<SWidget> PostInnerErrors(IErrorRef const& error) { return {}; };
41 };
