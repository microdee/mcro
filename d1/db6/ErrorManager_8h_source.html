23 /** @brief Control how an error is being displayed. Use C++ 20 designated initializers for convenience */
24 struct FDisplayErrorArgs
25 {
26 /**
27 * @brief
28 * The error message will not block the engine tick. This is useful for errors happening in the editor
29 * so even if PIE session is aborted due to an error, the developer can cross-check their assets with the
30 * error still open.
31 */
32 bool bAsync = false;
33
34 /** @brief Enables an extra checkbox which reminds the user to please do not immediately dismiss the error */
35 bool bImportantToRead = false;
36
37 /**
38 * @brief
39 * Set this to false if for any reason you don't need the debugger to break before displaying the error
40 * to the user.
41 */
42 bool bBreakDebugger = true;
43
44 /**
45 * @brief
46 * Set this to false if for any reason you don't need the error to be logged before displaying it to the user.
47 */
48 bool bLogError = true;
49
50 /**
51 * @brief
52 * Optionally set a parent widget for the modal window of the error. By default if not specified here the
53 * main editor window is used, or the main gameplay viewport.
54 */
55 TSharedPtr<const SWidget> Parent = {};
56 };

91 /**
92 * @brief
93 * Display the error summary for the user. Only use this when your program arrives to an unrecoverable state
94 * which either needs explanation for the user or requires action from the user (like configuration changes).
95 *
96 * @important
97 * The modal window and the widgets representing the error will be created on the main thread, keep that in
98 * mind while making the widgets for the errors.
99 *
100 * @param error The input error
101 * @param args Simple arguments object for this function, use initializer list or C++ 20 designated initializer.
102 *
103 * @return
104 * A future telling that either the dialog has been displayed or how it has been suppressed. The future also
105 * gives an opportunity to block the calling thread until the user acknowledges the error.
106 *
107 * @remarks
108 * Unless `bAsync` is set in the arguments, calling this function from any thread will also block the main
109 * thread while the modal window containing the error is open. If the calling thread also needs to be blocked
110 * then simply wait on the returned future.
111 *
112 * @todo
113 * Add ability to let the user "ignore" errors, and continue execution, because they know better.
114 */
115 auto DisplayError(IErrorRef const& error, FDisplayErrorArgs const& args) -> TFuture<EDisplayErrorResult>;
116
117 TEventDelegate<void()> OnErrorDialogDismissed;

119 private:
120

122 auto InferParentWidget() -> TSharedPtr<const SWidget>;
123
124 TSharedPtr<SWindow> ModalWindow;
