23 /**
24 * @brief
25 * This struct holds the circumstances of the data change. It cannot be moved or copied and its lifespan is
26 * managed entirely by `TState`
27 */
28 template <typename T>
29 struct TChangeData
30 {
31 template <CDefaultInitializable = T>
32 TChangeData() : Next() {}
33
34 template <CMoveConstructible = T>
35 TChangeData(T&& value) : Next(Forward<T>(value)) {}
36
37 template <CCopyConstructible = T>
38 TChangeData(const TChangeData& from) : Next(from.Next), Previous(from.Previous) {}
39
40 template <CMoveConstructible = T>
41 TChangeData(TChangeData&& from) : Next(MoveTemp(from.Next)), Previous(MoveTemp(from.Previous)) {}
42
43 template <typename Arg>
44 requires (!CSameAs<Arg, TChangeData> && !CSameAs<Arg, T>)
45 TChangeData(Arg&& arg) : Next(Forward<Arg>(arg)) {}
46
47 template <typename... Args>
48 requires (sizeof...(Args) > 1)
49 TChangeData(Args&&... args) : Next(Forward<Args>(args)...) {}
50
51 T Next;
52 TOptional<T> Previous;
53 };

55 /** @brief Public API and base class for `TState` which shouldn't concern with policy flags or thread safety */
56 template <typename T>
57 struct IState : IStateTag
58 {
59 using Type = T;
60 using ReadLockVariant = TVariant<FReadScopeLock, FVoid>;
61 using WriteLockVariant = TVariant<FWriteScopeLock, FVoid>;
62
63 virtual ~IState() = default;
64
65 /**
66 * @brief
67 * Get the wrapped value if for some reason the conversion operator is not enough or deleted.
68 * Thread safety is not considered in this function, use `ReadLock` before `Get`, or use `GetOnAnyThread`
69 * which provides a read lock, if thread safety is a concern.
70 */
71 virtual T const& Get() const = 0;
72
73 /**
74 * @brief
75 * Set the wrapped value if for some reason the assignment operator is not enough or deleted. When thread
76 * safety is enabled Set will automatically lock this state for writing.
77 *
78 * @warning
79 * Setting this state from within its change listeners is prohibited and will trigger a check()
80 */
81 virtual void Set(T const& value) = 0;
82
83 /**
84 * @brief Modify this state via an l-value ref in a functor
85 *
86 * @param modifier The functor which modifies this value
87 *
88 * @param alwaysNotify
89 * Notify observers about the change even when the previous state is not different after the modification.
90 * This is only applicable when T is copyable, comparable, StorePrevious flag is set and AlwaysNotify flag is
91 * not set via policy.
92 */
93 virtual void Modify(TUniqueFunction<void(T&)>&& modifier, bool alwaysNotify = true) = 0;
94
95 protected:
96 template <CChangeListener<T> Function>
97 static auto DelegateValueArgument(Function const& onChange, EInvokeMode invokeMode = DefaultInvocation)
98 {
99 return [onChange](TChangeData<T> const& change)
100 {
101 if constexpr (CChangeNextOnlyListener<Function, T>)
102 onChange(change.Next);
103 if constexpr (CChangeNextPreviousListener<Function, T>)
104 onChange(change.Next, change.Previous);
105 };
106 }
107 public:
108
109 /** @brief Add a delegate which gets a `TChangeData<T> const&` if this state has been set. */
110 virtual FDelegateHandle OnChange(TDelegate<void(TChangeData<T> const&)> onChange, EInvokeMode invokeMode = DefaultInvocation) = 0;
111
112 /**
113 * @brief
114 * Add a function without object binding which either has one or two arguments with the following signature:
115 * `[](T const& next, [TOptional<T> const& previous])`
116 *
117 * Where the argument `previous` is optional (to have, not its type). The argument `previous` when it is
118 * present is TOptional because it may only have a value when StorePrevious policy is active and T is copyable.
119 */
120 template <CChangeListener<T> Function>
121 FDelegateHandle OnChange(Function const& onChange, EInvokeMode invokeMode = DefaultInvocation)
122 {
123 return OnChange(InferDelegate::From(DelegateValueArgument(onChange)), invokeMode);
124 }
125
126 /**
127 * @brief
128 * Add a function with an object binding which either has one or two arguments with the following signature:
129 * `[](T const& next, [TOptional<T> const& previous])`
130 *
131 * Where the argument `previous` is optional (to have, not its type). The argument `previous` when it is
132 * present is TOptional because it may only have a value when StorePrevious policy is active and T is copyable.
133 */
134 template <typename Object, CChangeListener<T> Function>
135 FDelegateHandle OnChange(Object&& object, Function const& onChange, EInvokeMode invokeMode = DefaultInvocation)
136 {
137 return OnChange(InferDelegate::From(Forward<Object>(object), DelegateValueArgument(onChange)), invokeMode);
138 }

140 /**
141 * @brief
142 * Given value will be stored in the state only if T is equality comparable and it differs from the current
143 * state value. If T is not equality comparable this function is equivalent to Set and always returns true.
144 *
145 * @return
146 * True if the given value was different from the previous state value. Always returns true when T is is not
147 * equality comparable.
148 */
149 virtual bool HasChangedFrom(const T& nextValue) = 0;
150
151 /** @brief Returns true if this state has ever been changed from its initial value given at construction. */
152 virtual bool HasEverChanged() const = 0;
153
154 /** @brief Equivalent to `TMulticastDelegate::Remove` */
155 virtual bool Remove(FDelegateHandle const& handle) = 0;
156
157 /** @brief Equivalent to `TMulticastDelegate::RemoveAll` */
158 virtual int32 RemoveAll(const void* object) = 0;

160 /**
161 * @brief
162 * If thread safety is enabled in DefaultPolicy, get the value with a bundled read-scope-lock. Otherwise the
163 * tuple returns an empty dummy struct as its second argument.
164 *
165 * Use C++17 structured binding for convenience:
166 * @code
167 * auto [value, lock] = MyState.GetOnAnyThread();
168 * @endcode
169 *
170 * @remarks
171 * Unlike the placeholder `auto` keyword, the structured binding `auto` keyword preserves reference qualifiers.
172 * See https://godbolt.org/z/jn918fKfd
173 *
174 * @return
175 * The lock is returned as TUniquePtr it's slightly more expensive because of ref-counting but it makes the API
176 * so much easier to use as TState can decide to return a real lock or just a dummy.
177 */
178 virtual TTuple<T const&, TUniquePtr<ReadLockVariant>> GetOnAnyThread() const = 0;
179
180 /**
181 * @brief Lock this state for reading for the current scope.
182 *
183 * @return
184 * The lock is returned as TUniquePtr it's slightly more expensive because of ref-counting but it makes the API
185 * so much easier to use as TState can decide to return a real lock or just a dummy.
186 */
187 virtual TUniquePtr<ReadLockVariant> ReadLock() const = 0;
188
189 /**
190 * @brief Lock this state for writing for the current scope.
191 *
192 * @return
193 * The lock is returned as TUniquePtr it's slightly more expensive because of ref-counting but it makes the API
194 * so much easier to use as TState can decide to return a real lock or just a dummy.
195 */
196 virtual TUniquePtr<WriteLockVariant> WriteLock() = 0;

222 /**
223 * @brief
224 * Storage wrapper for any value which state needs to be tracked or their change needs to be observed.
225 * By default, TState is not thread-safe unless ThreadSafeState policy is active in DefaultPolicy
226 */
227 template <typename T, int32 DefaultPolicy>
228 struct TState : IState<T>
229 {
230 template <typename ThreadSafeType, typename NaiveType>
231 using ThreadSafeSwitch = std::conditional_t<static_cast<bool>(DefaultPolicy & ThreadSafeState), ThreadSafeType, NaiveType>;
232
233 using StateBase = IState<T>;
234
235 using typename StateBase::ReadLockVariant;
236 using typename StateBase::WriteLockVariant;
237
238 using ReadLockType = ThreadSafeSwitch<FReadScopeLock, FVoid>;
239 using WriteLockType = ThreadSafeSwitch<FWriteScopeLock, FVoid>;
240
241 static constexpr int32 DefaultPolicyFlags = DefaultPolicy;
242
243 /** @brief Enable default constructor only when T is default initializable */
244 template <CDefaultInitializable = T>
245 TState() : Value() {}
246
247 /** @brief Enable copy constructor for T only when T is copy constructable */
248 template <CCopyConstructible = T>
249 TState(T const& value) : Value(value) {}
250
251 /** @brief Enable move constructor for T only when T is move constructable */
252 template <CMoveConstructible = T>
253 TState(T&& value) : Value(MoveTemp(value)) {}
254
255 /** @brief Enable copy constructor for the state only when T is copy constructable */
256 template <CCopyConstructible = T>
257 TState(TState const& other) : Value(other.Value.Next) {}
258
259 /** @brief Enable move constructor for the state only when T is move constructable */
260 template <CMoveConstructible = T>
261 TState(TState&& other) : Value(MoveTemp(other.Value.Next)) {}
262
263 /** @brief Construct value in-place with non-semantic single argument constructor */
264 template <typename Arg>
265 requires (!CConvertibleTo<Arg, TState> && !CSameAs<Arg, T>)
266 TState(Arg&& arg) : Value(Forward<Arg>(arg)) {}
267
268 /** @brief Construct value in-place with multiple argument constructor */
269 template <typename... Args>
270 requires (sizeof...(Args) > 1)
271 TState(Args&&... args) : Value(Forward<Args>(args)...) {}

273 virtual T const& Get() const override { return Value.Next; }
274
275 virtual TTuple<T const&, TUniquePtr<ReadLockVariant>> GetOnAnyThread() const override
276 {
277 return { Value.Next, ReadLock() };
278 }
279
280 virtual void Set(T const& value) override
281 {
282 ASSERT_QUIT(!Modifying, ,
283 ->WithMessage(TEXT_"Attempting to set this state while this state is already being set from somewhere else.")
284 );
285 TGuardValue modifyingGuard(Modifying, true);
286 auto lock = WriteLock();
287 bool broadcast = true;
288
289 if constexpr (CCoreEqualityComparable<T>)
290 broadcast = PolicyFlags & AlwaysNotify || Value.Next != value;
291
292 if constexpr (CCopyable<T>)
293 if (PolicyFlags & StorePrevious)
294 Value.Previous = Value.Next;
295
296 Value.Next = value;
297
298 if (broadcast)
299 OnChangeEvent.Broadcast(Value);
300 }
301
302 virtual void Modify(TUniqueFunction<void(T&)>&& modifier, bool alwaysNotify = true) override
303 {
304 ASSERT_QUIT(!Modifying, ,
305 ->WithMessage(TEXT_"Attempting to set this state while this state is already being set from somewhere else.")
306 );
307 TGuardValue modifyingGuard(Modifying, true);
308 auto lock = WriteLock();
309 bool broadcast = true;
310
311 if constexpr (CCopyable<T>)
312 {
313 if (PolicyFlags & StorePrevious)
314 {
315 Value.Previous = Value.Next;
316 }
317 }
318
319 modifier(Value.Next);
320
321 if constexpr (CCopyable<T> && CCoreEqualityComparable<T>)
322 broadcast = alwaysNotify
323 || !(PolicyFlags & StorePrevious)
324 || PolicyFlags & AlwaysNotify
325 || !Value.Previous.IsSet()
326 || Value.Previous.GetValue() != Value.Next;
327
328 if (broadcast)
329 OnChangeEvent.Broadcast(Value);
330 }
331
332 virtual FDelegateHandle OnChange(TDelegate<void(TChangeData<T> const&)> onChange, EInvokeMode invokeMode = DefaultInvocation) override
333 {
334 auto lock = WriteLock();
335 return OnChangeEvent.Add(onChange, invokeMode);
336 }
337
338 virtual bool Remove(FDelegateHandle const& handle) override
339 {
340 auto lock = WriteLock();
341 return OnChangeEvent.Remove(handle);
342 }
343
344 virtual int32 RemoveAll(const void* object) override
345 {
346 auto lock = WriteLock();
347 return OnChangeEvent.RemoveAll(object);
348 }
349
350 virtual bool HasChangedFrom(const T& nextValue) override
351 {
352 if constexpr (CCoreEqualityComparable<T>)
353 {
354 bool hasChanged = Value.Next != nextValue;
355 Set(nextValue);
356 return hasChanged;
357 }
358 else
359 {
360 Set(nextValue);
361 return true;
362 }
363 }
364
365 virtual bool HasEverChanged() const override
366 {
367 return OnChangeEvent.IsBroadcasted();
368 }
369
370 virtual TUniquePtr<ReadLockVariant> ReadLock() const override
371 {
372 return MakeUnique<ReadLockVariant>(TInPlaceType<ReadLockType>(), Mutex.Get());
373 }
374
375 virtual TUniquePtr<WriteLockVariant> WriteLock() override
376 {
377 return MakeUnique<WriteLockVariant>(TInPlaceType<WriteLockType>(), Mutex.Get());
378 }

380 int32 PolicyFlags = DefaultPolicy;

382 private:
383 TEventDelegate<void(TChangeData<T> const&)> OnChangeEvent;
384 TChangeData<T> Value;
385 bool Modifying = false;
386 mutable TInitializeOnCopy<FRWLock> Mutex;
387 };
